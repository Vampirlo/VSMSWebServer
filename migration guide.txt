узнайте версию 
dotnet list package | findstr Microsoft.EntityFrameworkCore

установите пакет нужной версии 
dotnet add package Microsoft.EntityFrameworkCore.Design --version 9.0.8

если не той устновили, то удалить
dotnet remove package Microsoft.EntityFrameworkCore.Design

создайте миграцию
dotnet ef migrations add InitialCreate

Примените миграцию к базе данных
dotnet ef database update
